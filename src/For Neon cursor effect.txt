import React, { useRef, useState, useEffect } from 'react'
import Navbar from './Components/Navbar'
import Hero from './Components/Hero'
import TrustedBy from './Components/TrustedBy'
import Services from './Components/Services'
import OurWork from './Components/OurWork'
import Teams from './Components/Teams'
import Contacts from './Components/Contacts'
import { Toaster } from 'react-hot-toast'
import Footer from './Components/Footer'

function App() {

  const [theme, setTheme] = useState(
    localStorage.getItem('theme') ? localStorage.getItem('theme') : 'light'
  )

  const dotRef = useRef(null)
  const outlineRef = useRef(null)

  // Cursor position refs
  const mouse = useRef({ x: 0, y: 0 })
  const position = useRef({ x: 0, y: 0 })

  useEffect(() => {
    const handleMouseMove = (e) => {
      mouse.current.x = e.clientX
      mouse.current.y = e.clientY
    }

    document.addEventListener('mousemove', handleMouseMove)

    const speed = 0.25 // ðŸ”¥ increase = faster outer circle

    const animate = () => {
      position.current.x += (mouse.current.x - position.current.x) * speed
      position.current.y += (mouse.current.y - position.current.y) * speed

      if (dotRef.current && outlineRef.current) {
        // Cursor dot
        dotRef.current.style.transform = `translate3d(
          ${mouse.current.x - 6}px,
          ${mouse.current.y - 6}px,
          0
        )`

        // Neon outer ring
        outlineRef.current.style.transform = `translate3d(
          ${position.current.x - 24}px,
          ${position.current.y - 24}px,
          0
        )`

        outlineRef.current.style.boxShadow = `
          0 0 20px rgba(80, 68, 229, 0.6),
          0 0 40px rgba(80, 68, 229, 0.4),
          0 0 80px rgba(80, 68, 229, 0.25)
        `
      }

      requestAnimationFrame(animate)
    }

    animate()

    return () => {
      document.removeEventListener('mousemove', handleMouseMove)
    }
  }, [])

  return (
    <div className="dark:bg-black relative">
      <Toaster />
      <Navbar theme={theme} setTheme={setTheme} />
      <Hero />
      <TrustedBy />
      <Services />
      <OurWork />
      <Teams />
      <Contacts />
      <Footer theme={theme} />

      {/* Neon Cursor Ring */}
      <div
        ref={outlineRef}
        className="
          fixed top-0 left-0
          h-12 w-12
          rounded-full
          border border-primary
          bg-primary/10
          blur-[2px]
          pointer-events-none
          z-[9999]
        "
        style={{
          animation: 'neonPulse 1.8s infinite ease-in-out',
          background:
            'radial-gradient(circle, rgba(80,68,229,0.35) 0%, transparent 70%)'
        }}
      />

      {/* Cursor Dot */}
      <div
        ref={dotRef}
        className="
          fixed top-0 left-0
          h-3 w-3
          rounded-full
          bg-primary
          pointer-events-none
          z-[9999]
        "
      />
    </div>
  )
}

export default App
